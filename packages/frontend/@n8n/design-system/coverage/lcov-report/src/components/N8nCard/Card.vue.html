<script lang="ts" setup>
import { computed, useCssModule } from 'vue';
 
interface CardProps {
	hoverable?: boolean;
}
 
defineOptions({ name: 'N8nCard' });
const props = withDefaults(defineProps<CardProps>(), {
	hoverable: false,
});
 
const $style = useCssModule();

	card: true,
	[$style.card]: true,
	[$style.hoverable]: props.hoverable,

	</div>
</template>
